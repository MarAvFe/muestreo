exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO Sampling (idSampling, description, live, name, SamplingType_idSamplingType, estimatedTrailTime_Minutes, isPreliminarSampling, p_preliminar, q_preliminar, error_preliminar, n_preliminar, z_preliminar, p_definitive, q_definitive, error_definitive, n_definitive, z_definitive, abs_precision_preliminar, abs_precision_definitive, relative_precision_preliminar, relative_precision_definitive) VALUES( :V_idSampling, :V_description, :V_live, :V_name, :V_SamplingType_idSamplingType, :V_estimatedTrailTime_Minutes, :V_isPreliminarSampling, :V_p_preliminar, :V_q_preliminar, :V_error_preliminar, :V_n_preliminar, :V_z_preliminar, :V_p_definitive, :V_q_definitive, :V_error_definitive, :V_n_definitive, :V_z_definitive, :V_abs_precision_preliminar, :V_abs_precision_definitive, :V_relative_precision_preliminar, :V_relative_precision_definitive )", req.body, res);
	});
 	router.use("/Sampling", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM Sampling WHERE :OU", req.body, res);
	});
 	router.use("/Sampling", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE Sampling SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/Sampling", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM Sampling WHERE :OR", req.body, res);
	});
 	router.use("/Sampling", deletePath);

	return router;
}
