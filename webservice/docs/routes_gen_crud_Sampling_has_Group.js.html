	});
 	router.use("/Sampling_has_Group", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM Sampling_has_Group WHERE :OU", req.body, res);
	});
 	router.use("/Sampling_has_Group", read);

	var update = express.Router();

		connection.query("UPDATE Sampling_has_Group SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/Sampling_has_Group", update);

	var deletePath = express.Router();

		connection.query("DELETE FROM Sampling_has_Group WHERE :OR", req.body, res);
