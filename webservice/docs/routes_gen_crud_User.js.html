var express = require("express");


/**
 * @function User
 *@param {int(11)} idUser - Key: PRI
 *@param {varchar(12)} cedula
 *@param {varchar(45)} name
 *@param {varchar(45)} lastname
 *@param {varchar(255)} email
 *@param {varchar(12)} phone
 *@param {varchar(45)} pwd
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO User (idUser, cedula, name, lastname, email, phone, pwd) VALUES( :V_idUser, :V_cedula, :V_name, :V_lastname, :V_email, :V_phone, :V_pwd )", req.body, res);
	});
 	router.use("/User", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM User WHERE :OU", req.body, res);
	});
 	router.use("/User", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE User SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/User", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM User WHERE :OR", req.body, res);
	});
 	router.use("/User", deletePath);

	return router;
}
