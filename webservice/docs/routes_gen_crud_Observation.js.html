var express = require("express");

 *@param {bit(1)} hasData
 *@param {bit(1)} isProductive
 *@param {bit(1)} isCancelled
 *@param {int(11)} Worker_idWorker - Key: PRI
 *@param {int(11)} User_idUser - Key: PRI
 *@param {int(11)} Trail_idTrail - Key: PRI
 *@param {int(11)} ImprodAct_idImprodAct - Key: PRI
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO Observation (idObservation, date, hasData, isProductive, isCancelled, Worker_idWorker, User_idUser, Trail_idTrail, ImprodAct_idImprodAct) VALUES( :V_idObservation, :V_date, :V_hasData, :V_isProductive, :V_isCancelled, :V_Worker_idWorker, :V_User_idUser, :V_Trail_idTrail, :V_ImprodAct_idImprodAct )", req.body, res);
	});
 	router.use("/Observation", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM Observation WHERE :OU", req.body, res);
	});
 	router.use("/Observation", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE Observation SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/Observation", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM Observation WHERE :OR", req.body, res);
	});
 	router.use("/Observation", deletePath);

	return router;
}
