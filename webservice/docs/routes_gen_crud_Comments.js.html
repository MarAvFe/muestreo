var express = require("express");


/**
 * @function Comments
 *@param {int(11)} idComments - Key: PRI
 *@param {varchar(500)} comment
 *@param {date} date
 *@param {int(11)} User_idUser - Key: PRI
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO Comments (idComments, comment, date, User_idUser) VALUES( :V_idComments, :V_comment, :V_date, :V_User_idUser )", req.body, res);
	});
 	router.use("/Comments", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM Comments WHERE :OU", req.body, res);
	});
 	router.use("/Comments", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE Comments SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/Comments", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM Comments WHERE :OR", req.body, res);
	});
 	router.use("/Comments", deletePath);

	return router;
}
