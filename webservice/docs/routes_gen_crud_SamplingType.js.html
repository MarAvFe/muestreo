var express = require("express");


/**
 * @function SamplingType
 *@param {int(11)} idSamplingType - Key: PRI
 *@param {varchar(45)} name
 *@param {varchar(4)} initials
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO SamplingType (idSamplingType, name, initials) VALUES( :V_idSamplingType, :V_name, :V_initials )", req.body, res);
	});
 	router.use("/SamplingType", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM SamplingType WHERE :OU", req.body, res);
	});
 	router.use("/SamplingType", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE SamplingType SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/SamplingType", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM SamplingType WHERE :OR", req.body, res);
	});
 	router.use("/SamplingType", deletePath);

	return router;
}
