var express = require("express");


/**
 * @function ImprodAct
 *@param {int(11)} idImprodAct - Key: PRI
 *@param {varchar(45)} name
 *@param {varchar(255)} description
 *@param {bit(1)} isCollaborative
 */


exports.router = function (connection) {
	var router = express.Router();

	var create = express.Router();
	create.post("/add", function (req, res) {
		connection.query("INSERT INTO ImprodAct (idImprodAct, name, description, isCollaborative) VALUES( :V_idImprodAct, :V_name, :V_description, :V_isCollaborative )", req.body, res);
	});
 	router.use("/ImprodAct", create);

	var read = express.Router();
	read.post("/get", function (req, res) {
		connection.query("SELECT :C FROM ImprodAct WHERE :OU", req.body, res);
	});
 	router.use("/ImprodAct", read);

	var update = express.Router();
	update.post("/update", function (req, res) {
		connection.query("UPDATE ImprodAct SET :OC WHERE :OF", req.body, res);
	});
 	router.use("/ImprodAct", update);

	var deletePath = express.Router();
	deletePath.post("/delete", function (req, res) {
		connection.query("DELETE FROM ImprodAct WHERE :OR", req.body, res);
	});
 	router.use("/ImprodAct", deletePath);

	return router;
}
